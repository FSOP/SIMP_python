CREATE TABLE decay (
norad_cat_id int, 
object_number int,
object_name string,
intldes string,
object_id string,
rcs double,
rcs_size string,
country string,
msg_epoch datetime,
decay_epoch datetime,
source string,
msg_type string,
precedence string,
primary key(norad_cat_id, msg_epoch)
)
